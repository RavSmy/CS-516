using namespace std;

int fib(int n); 

int main() {
	for (int i = 0; i < 4; i++) {
		int n;
		cin >> n;
		cout << "Answer: " << fib(n) << "\n";
	}
}

int fib(int n) {
	if (n == 0) return 0;
	else if (n == 1) return 1; 
	
	return fib(n - 1) + fib(n - 2);
}
