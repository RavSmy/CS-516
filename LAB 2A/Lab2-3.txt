void tenoct(int oct[]) {
	oct[3] += 1;
	if (oct[3] > 7) { 
		oct[3] = oct[3] - 8;
		oct[2]++;

		if (oct[2] > 7) {
			oct[2] = oct[2] - 8;
			oct[1]++;

			if (oct[1] > 7) {
			oct[1] = oct[1] - 8;
			oct[0]++;
			}
		}

	}
	cout << oct[0] << oct[1] << oct[2] << oct[3] << "\n"; 
}
